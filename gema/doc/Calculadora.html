def impactoAmbientalGEI(menu)
        aux = 0
        menu.each do |i|
        aux = aux + i.gei
        end
        return aux.round(2)
end

def valorKiloCalorico(comida)
        aux = ((comida.proteinas * 4) + (comida.lipidos * 9) + (comida.carbohidratos * 4))
        return aux.round(2)   
end
